string getNames() {
	string name;
	cout << "Enter a students first and last name: ";
	getline(cin, name);
	return name;
}
float getGPAs() {
	float gpa;
	cout << "Enter the student's gpa: ";
	cin >> gpa;
	cin.ignore(200, '\n');
	return gpa;
}

int main() {

	cout << "Welcome to Wacky Winners" << endl;

	const int NUM_STUDENTS = 5;
	const int COLM1 = 30;
	const int COLM2 = 10;

	//Declare Arrays
	string students[NUM_STUDENTS];
	float gpa[NUM_STUDENTS];

	//initialize best and worst student positions in arrays for comparisons

	float gpaHigh = 0;
	float gpaLow = 5;

	int lowIndex;
	
	
	

	//Prompt users for names and grades
	for (int i = 0; i < NUM_STUDENTS; i++) {
		students[i] = getNames();
		gpa[i] = getGPAs();
		cout << endl;

		//compare the GPAs of the students and record the best and worst GPA
		if (gpa[i] > gpaHigh) {
			gpaHigh = gpa[i];
			highIndex = i;
		}
		if (gpa[i] < gpaLow) {
			gpaLow = gpa[i];
			lowIndex = i;
		}
	}

	}
